%% Set up fancy headers and footers.
\usepackage{fancyhdr}
\pagestyle{empty}

\fancypagestyle{front}{
  \fancyhf{}
  \ifdefined\ONLINEVERSION
    \fancyhead[L]{\sffamily\nouppercase{\leftmark}}
  \else
    \fancyhead[LE]{\sffamily\nouppercase{\leftmark}}
    \fancyhead[RO]{\sffamily\doctitle}
  \fi
  \fancyfoot[C]{\thepage}
  \fancypagestyle{plain}{%
    \renewcommand{\headrulewidth}{0pt}%
    \fancyhead{}%
    \fancyfoot[C]{\thepage}%
  }
}
\fancypagestyle{main}{
  \fancyhf{}
  \ifdefined\ONLINEVERSION
    \fancyhead[L]{\sffamily\nouppercase{\leftmark}}
  \else
    \fancyhead[LE]{\sffamily\nouppercase{\leftmark}}
    \fancyhead[RO]{\sffamily\doctitle}
  \fi
  \fancyfoot[C]{\thepage~/~\pageref{LastPage}}
  \fancypagestyle{plain}{%
    \renewcommand{\headrulewidth}{0pt}%
    \fancyhead{}%
    \fancyfoot[C]{\thepage~/~\pageref{LastPage}}%
  }
}

\NewCommandCopy{\frontmatterOrig}{\frontmatter}
\NewCommandCopy{\mainmatterOrig}{\mainmatter}
\NewCommandCopy{\backmatterOrig}{\backmatter}

\renewcommand{\frontmatter}{\frontmatterOrig\pagestyle{front}}
\renewcommand{\mainmatter}{\mainmatterOrig\pagestyle{main}}
\renewcommand{\backmatter}{\backmatterOrig\pagestyle{main}}

%% Set up chapter heading style.

% Extract the leading character.
\ExplSyntaxOn
\cs_generate_variant:Nn \tl_item:nn { f }
\NewDocumentCommand{\ExtractLeadingChars}{O{1} m}{
  \tl_item:fn { #2 } { #1 }
}
\ExplSyntaxOff

% Draw the chapter heading letter in the background.
\NewDocumentCommand{\chapterheadingletter}{m}{%
  \makebox[0pt][l]{%
    \raisebox{-16pt}[0pt][0pt]{%
      \hspace{0.55em}\color{gray!50}%
      \usefont{T1}{qzc}{m}{it}\fontsize{95pt}{95pt}\selectfont%
      #1 %
    }%
  }%
}

% Chapter heading formatter.
\NewDocumentCommand{\chaphead}{s O{} m}{
  \vspace*{-25pt}%
  {\setlength{\parindent}{0pt}\raggedright
    \Huge\sffamily\bfseries
    \chapterheadingletter{#2}%
    #3\par\nobreak%
    \vspace{40pt}%
  }
}

% Redefine commands to use the new chapter heading style.
\makeatletter
\def\@makechapterhead#1{\chaphead[\thechapter]{#1}}
\def\@makeschapterhead#1{\chaphead*[\ExtractLeadingChars{#1}]{#1}}
\makeatother

% macros for title with no number
\newcommand{\chapternotnumbered}[1]{%
  \chapter*{#1}%
  \markboth{#1}{#1}%
  \addcontentsline{toc}{chapter}{#1}%
}

%% Set up title styles.
\usepackage{titlesec}

% section
\titleformat{\section}
  {\sffamily\bfseries\large}
  {\thesubsection}
  {0.5em}
  {}

% subsection
\titleformat{\subsection}
  {\sffamily\bfseries\centering}
  {}
  {0pt}
  {}

%% Set up toc styles.
\usepackage{xcolor}
\usepackage{titletoc}

\setcounter{tocdepth}{2}

% chapter
\titlecontents{chapter}[1.6em]
  {\addvspace{0.5em}\bfseries}
  {\contentslabel{1.4em}}{\hspace*{-1.4em}}
  {\hfill\normalfont\contentspage[\bfseries\thecontentspage]}

% section
\titlecontents{section}[4.0em]
  {\addvspace{0.2em}}
  {\contentslabel{2.3em}}{\hspace*{-2.3em}}
  {\textcolor{gray}{\titlerule*[9pt]{.}}\contentspage}

% subsection
\newcommand*{\sspace}{\hspace{0.34em plus 0.5em minus 0.12em}}
\titlecontents*{subsection}[4.7em]
  {\footnotesize\color{gray}}
  {}{}
  {~\thecontentspage}
  []
  [\sspace\textbullet\sspace][\hspace*{0.6em}\vspace{0.1em}]

% Custom macro to show the contents.
\newcommand{\makecontents}{%
  \hypersetup{hidelinks}%

}
